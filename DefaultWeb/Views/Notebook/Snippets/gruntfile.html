  grunt.registerTask('buildswatch', 'build a regular theme', function (theme) {

      var theme = theme == undefined ? grunt.config('buildtheme') : theme;

 

      grunt.config.buildtheme = theme;

 

    

      var themedir = './Content/bootswatch/' + theme;

      grunt.log.debug(themedir);

 

      var isValidTheme = grunt.file.exists(themedir, 'variables.less') && grunt.file.exists(themedir, 'bootswatch.less');

 

      // cancel the build (without failing) if this directory is not a valid theme

      if (!isValidTheme) {

          grunt.log.debug(themedir + ' is invalid!');

          return;

      }

 

      grunt.config.buildtheme = theme;

 

      var concatSrc;

      var concatDest;

      var lessDest;

      var lessSrc;

      var files = {};

      var dist = {};

      concatSrc = './Content/bootswatch/global/build.less';

      concatDest = themedir + '/build.less';

      lessSrc = themedir + '/build.less';

      lessDest = './Content/style/themes/' + theme + '.css';

      var lessminDest = './Content/style/themes/' + theme + '.min.css';

     

 

      dist = { src: concatSrc, dest: concatDest };

      grunt.config('concat.dist', dist);

      grunt.task.run('concat:dist');

 

      files = {};

      files[lessDest] = lessSrc;

      grunt.config('less.dist.files', files);

      grunt.task.run('less:dist');

 

      grunt.config('autoprefixer.dist.src', lessDest);

      grunt.task.run('autoprefixer:dist');

 

      dist = { src: lessDest, dest: lessminDest };

      grunt.config('cssmin.minifyTheme', dist);

      grunt.task.run('cssmin:minifyTheme');

  });
