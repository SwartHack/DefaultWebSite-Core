public ActionResult Index(string virtualPath)

{

    ArchiveFile file  = MvcApplication.Archive.GetFile(HttpUtility.UrlDecode(virtualPath));

    if (file != null)

    {

        var fileJson = GetJsonFile(file);

 

        Response.StatusCode = 200;

        if (Request.IsAjaxRequest())

        {

            return Json(fileJson, JsonRequestBehavior.AllowGet);

        }

        MvcApplication.Folders = file.ParentFolder.SubFolders;

        MvcApplication.SelectedFolder = file.ParentFolder;

        return View("~/Main/Index");

    }

    else

    {

        Response.StatusCode = 404;

        return Content("File not found at " + virtualPath);

    }

}
