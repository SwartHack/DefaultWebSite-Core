namespace ArchiveWeb

{

    public class Program

    {

        public static void Main(string[] args)

        {

            var host = new WebHostBuilder()

                .UseKestrel()

                .UseContentRoot(Directory.GetCurrentDirectory())

                .UseIISIntegration()

                .UseKestrel()

                .UseStartup<Startup>()

                .UseApplicationInsights()

                .Build();

 

            host.Run();

        }

    }

}
