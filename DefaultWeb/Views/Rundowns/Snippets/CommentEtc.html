namespace DefaultWeb2.Models.Entities

{

    public class Comment

    {

        public string ID { get; set; }

        /// <summary>

        /// Text decorations for JQuery Validate

        /// </summary>

        [Required(ErrorMessage = "Author is a required field")]

        [StringLength(100, ErrorMessage = "Value too long")]

        public string Author { get; set; }

 

        [ScriptIgnore] //we don't want to expose this

        public string Email { get; set; }

 

        [Required(ErrorMessage = "Please enter a body")]

        [StringLength(300, ErrorMessage = "Value too long")]

        public string Body { get; set; }

 

        public DateTime TimeStamp { get; set; }

 

        [Required]

        [ScriptIgnore]

        public string Location { get; set; }

 

        [Required]

        public string SocialIcon { get; set; }

 

        [XmlIgnoreAttribute]

        [ScriptIgnore]

        public string Type { get; set; }

 

        public string ThumbnailFullUrl

        {

            get

            {

                return string.Format("{0}{1}{2}", MvcApplication.VirtualRootPath, "/thumbnailview/60/60?virtualpath=", HttpUtility.UrlEncode(Location));

            }

        }

 

        public bool CanDeleteComment

        {

            get

            {

                if (HttpContext.Current == null) // in unit testing.

                    return false;

 

                bool isUserLogged = HttpContext.Current.Request.IsAuthenticated;

                bool isAdmin = isUserLogged && HttpContext.Current.User.IsInRole("Admin");

                bool isCommentOwner = isUserLogged && HttpContext.Current.User.Identity.Name == Email;

                bool isCommentRecent = TimeStamp.AddMinutes(20) > DateTime.Now;

 

                return isAdmin || (isCommentOwner && isCommentRecent);

            }

        }

       .

       .

       .

    }

}
