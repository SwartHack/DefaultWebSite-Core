/// <summary>

/// /

/// </summary>

/// <param name="file"></param>

/// <returns></returns>

public object GetJsonFile(ArchiveFile file)

{

    var link1 = HttpUtility.UrlEncode(String.Format("{0}://{1}/fileview?virtualpath={2}", Request.Url.Scheme, Request.Url.Host, HttpUtility.UrlPathEncode(file.VirtualPath)));

 

    var fileJson = new

    {

        Name = file.FileName,

        FriendlyName = file.FriendlyName,

        Description = file.Description,

        VirtualPath = file.VirtualPath,

        ViewURL = file.FileViewUrl,

        FileURL = file.FileMainUrl,

        FullURL = link1,

        Comments = file.GetFileComments(),

        ThumbnailURL = file.ThumbnailUrl,

        MimeType = file.MimeType[1],

        EXIF = file.getEXIF()

    };

    return fileJson;

}
