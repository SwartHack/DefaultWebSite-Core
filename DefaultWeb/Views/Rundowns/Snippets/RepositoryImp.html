namespace DefaultWeb2.Models.Repositories

{

    /// <summary>

    /// Archive repository. New instance per request.

    /// </summary>

    ///

    public class ArchiveRepository : iArchiveRepository

    {

        private string _relativeAppVirtualUrl;

        private static object cacheLock = new object();

 

        public ArchiveFolder RootFolder { get; set; }

        public List<ArchiveFolder> Folders { get; set; }

        //public CommentRepository CommentRepository { get; set; }

       

        public string PhysicalPath { get; set; }

 

        public ArchiveRepository(string basePhysicalPath)

        {

            if (!Directory.Exists(basePhysicalPath)) {

                throw new ConfigurationException("Invalid path: " + basePhysicalPath);

            }

 

            PhysicalPath = new DirectoryInfo(basePhysicalPath).FullName;

 

            if (string.IsNullOrWhiteSpace(MvcApplication.VirtualRootPath))

            {

                MvcApplication.VirtualRootPath = "~/";

            }

 

            _relativeAppVirtualUrl = MvcApplication.VirtualRootPath;

            //CommentRepository = new CommentRepository(this);

 

            RootFolder = new ArchiveFolder();

            RootFolder.PhysicalPath = basePhysicalPath;

 

            FillSubFoldersRecursive(RootFolder, new ArchiveFolderNameASCComparer());

 

            Folders = RootFolder.SubFolders;

        }

   

        /// <summary>

        /// Gets repository folder based on virtual Path

        /// </summary>

        /// <param name="folderVirtualPath"></param>

        /// <returns>ArchiveFolder instance or null if not found</returns>

        public ArchiveFolder GetFolder(string folderVirtualPath)

        {

            foreach (ArchiveFolder folder in Folders) {

                ArchiveFolder found = folder.SearchSubFolderRecursiveByVPath(folderVirtualPath);

                if (found != null) {

                    return found;

                }

            }

            return null;

        }

 

        /// <summary>

        /// Get file from virtual path

        /// </summary>

        /// <param name="fileVirtualPath"></param>

        /// <returns>ArchiveFile instance or null if not found</returns>

        public ArchiveFile GetFile(string fileVirtualPath)

        {

            foreach (ArchiveFile file in GetAllFiles(RootFolder))

            {

                if (Util.IsSameVpath(file.VirtualPath, fileVirtualPath))

                {

                    return file;

                }

            }

            return null;

        }

 

        /// <summary>

        /// Returns all files in the archive under folder

        /// </summary>

        /// <returns></returns>

        public List<ArchiveFile> GetAllFiles(ArchiveFolder folder)

        {

 

            List<ArchiveFile> files = new List<ArchiveFile>();

 

            if (folder == null)

                folder = RootFolder;

 

            //process this folder, then subfolders. Recursive!!!

            files.AddRange(folder.Files);

            foreach (ArchiveFolder fld in folder.SubFolders)

            {

                files.AddRange(GetAllFiles(fld));

            }

            return files;

        }

 

        public List<Comment> GetComments(ArchiveFolder folder, string id)

        {

            List<Comment> comments = new List<Comment>();

 

            // if null folder, use Archive root folder ??

            if (folder == null)

                folder = RootFolder;

 

            foreach (ArchiveFile file in GetAllFiles(folder))

            {

                comments.AddRange(file.Comments);

            }

             

            if (!string.IsNullOrEmpty(id))

                return comments.Where(c => c.ID == id).ToList();

 

            return comments;

        }

 

        public void AddComment(Comment comment)

        {

            ArchiveFolder folder = null;

 

            switch (comment.Type)

            {

               

                case "FILE":

                    var file = GetFile(comment.Location);

                    if ( file == null)

                    {

                        //deal with this better TODO

                        throw new ArgumentException("file not found: comment.Location");

                    }

                    file.Comments.Add(comment);

                    folder = file.ParentFolder;

                    break;

 

                case "FOLDER":

                    folder = GetFolder(comment.Location);

                    if ( folder == null)

                    {

                        //deal with this better TODO

                        throw new ArgumentException("folder not found: comment.Location");

                    }

                    folder.Comments.Add(comment);

                    break;

 

                default:

                    //deal with this better TODO

                    //throw new ArgumentException("Unknown comment type:comment.Type");

                    return;

            }

 

            folder.WriteFolderFileInfo();

        }

 

        public void DeleteComment(Comment comment)

        {

            ArchiveFile file = GetFile(comment.Location);

            if (file == null)

                throw new ArgumentException("file not found");

 

            file.Comments.Remove(comment);

            file.ParentFolder.WriteFolderFileInfo();

        }

 

        public List<ArchiveFile> Search(string q) {

            return GetAllFiles(RootFolder).Where(

                i =>

                    (i.FriendlyName.IndexOf(q, StringComparison.InvariantCultureIgnoreCase) > -1)

                    ||

                    (i.Description.IndexOf(q, StringComparison.InvariantCultureIgnoreCase) > -1)

                    ||

                    (i.VirtualPath.IndexOf(q, StringComparison.InvariantCultureIgnoreCase) > -1)

                    ||

                    (i.ParentFolder != null && i.ParentFolder.Name != null && i.ParentFolder.Name.IndexOf(q, StringComparison.InvariantCultureIgnoreCase) > -1)

                ).ToList();   

        }

        {
